#include <stdio.h>
#include <stdlib.h>

struct node {
    int data;
    struct node* next;
};

// Display function
void display(struct node* head) {
    while (head != NULL) {
        printf("%d -> ", head->data);
        head = head->next;
    }
    printf("NULL\n");
}

int main() {
    struct node *head1 = NULL, *head2 = NULL, *temp;

    // First list: 1 -> 2 -> 3
    head1 = (struct node*)malloc(sizeof(struct node));
    head1->data = 1;
    head1->next = (struct node*)malloc(sizeof(struct node));
    head1->next->data = 2;
    head1->next->next = (struct node*)malloc(sizeof(struct node));
    head1->next->next->data = 3;
    head1->next->next->next = NULL;

    // Second list: 4 -> 5 -> 6
    head2 = (struct node*)malloc(sizeof(struct node));
    head2->data = 4;
    head2->next = (struct node*)malloc(sizeof(struct node));
    head2->next->data = 5;
    head2->next->next = (struct node*)malloc(sizeof(struct node));
    head2->next->next->data = 6;
    head2->next->next->next = NULL;

    printf("List 1: ");
    display(head1);

    printf("List 2: ");
    display(head2);

    // Concatenation
    temp = head1;
    while (temp->next != NULL) {
        temp = temp->next;
    }
    temp->next = head2;

    printf("After Concatenation: ");
    display(head1);

    return 0;
}
