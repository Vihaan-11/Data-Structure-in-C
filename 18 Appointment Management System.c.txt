#include <stdio.h>
#include <stdlib.h>
#include <string.h>

#define MAX 100

struct Patient {
    int id;
    char name[50];
};

struct Patient queue[MAX];
int front = -1, rear = -1;

void addPatient() {
    if (rear == MAX - 1) {
        printf("Queue is full!\n");
        return;
    }

    struct Patient p;
    printf("Enter Patient ID: ");
    scanf("%d", &p.id);
    printf("Enter Patient Name: ");
    scanf("%s", p.name);

    if (front == -1) front = 0;
    queue[++rear] = p;

    printf("Patient added successfully!\n");
}

void viewNext() {
    if (front == -1 || front > rear) {
        printf("No patients in queue.\n");
        return;
    }

    printf("Next Patient -> ID: %d, Name: %s\n",
           queue[front].id, queue[front].name);
}

void servePatient() {
    if (front == -1 || front > rear) {
        printf("No patients to serve.\n");
        return;
    }

    printf("Serving Patient -> ID: %d, Name: %s\n",
           queue[front].id, queue[front].name);

    front++;

    if (front > rear) {
        front = rear = -1; // reset queue
    }
}

void displayPatients() {
    if (front == -1 || front > rear) {
        printf("No patients waiting.\n");
        return;
    }

    printf("\n--- Waiting Patients ---\n");
    for (int i = front; i <= rear; i++) {
        printf("ID: %d, Name: %s\n",
               queue[i].id, queue[i].name);
    }
}

int main() {
    int choice;

    while (1) {
        printf("\n--- Clinic Appointment System ---\n");
        printf("1. Add Patient\n");
        printf("2. View Next Patient\n");
        printf("3. Serve Patient\n");
        printf("4. Display All Patients\n");
        printf("5. Exit\n");
        printf("Enter your choice: ");
        scanf("%d", &choice);

        switch (choice) {
            case 1: addPatient(); break;
            case 2: viewNext(); break;
            case 3: servePatient(); break;
            case 4: displayPatients(); break;
            case 5: exit(0);
            default: printf("Invalid choice!\n");
        }
    }

    return 0;
}
