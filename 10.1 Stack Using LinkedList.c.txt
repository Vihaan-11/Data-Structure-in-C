#include <stdio.h>
#include <stdlib.h>

struct node {
    int data;
    struct node* next;
};

struct node* top = NULL;

// Push
void push(int x) {
    struct node* n = (struct node*)malloc(sizeof(struct node));
    n->data = x;
    n->next = top;
    top = n;
}

// Pop
void pop() {
    if (top == NULL) {
        printf("Stack Underflow\n");
        return;
    }
    struct node* temp = top;
    printf("Deleted: %d\n", temp->data);
    top = top->next;
    free(temp);
}

// Display
void display() {
    struct node* temp = top;
    while (temp != NULL) {
        printf("%d -> ", temp->data);
        temp = temp->next;
    }
    printf("NULL\n");
}

int main() {
    push(10);
    push(20);
    push(30);
    display();
    pop();
    display();
    return 0;
}
