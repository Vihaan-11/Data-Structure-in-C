#include <stdio.h>
#define MAX 100

char stack[MAX];
int top = -1;

void push(char x) {
    stack[++top] = x;
}

char pop() {
    return stack[top--];
}

int priority(char x) {
    if (x == '(')
        return 0;
    if (x == '+' || x == '-')
        return 1;
    if (x == '*' || x == '/')
        return 2;
    return 3;
}

int isOperand(char ch) {
    if ((ch >= 'A' && ch <= 'Z') ||
        (ch >= 'a' && ch <= 'z') ||
        (ch >= '0' && ch <= '9'))
        return 1;
    return 0;
}

int main() {
    char infix[MAX], postfix[MAX];
    int i = 0, j = 0;
    char x;

    printf("Enter infix expression: ");
    scanf("%s", infix);

    push('(');

    while (infix[i] != '\0')
        i++;
    infix[i] = ')';
    infix[i + 1] = '\0';

    i = 0;

    while (infix[i] != '\0') {

        if (infix[i] == '(')
            push(infix[i]);

        else if (isOperand(infix[i]))
            postfix[j++] = infix[i];

        else if (infix[i] == ')') {
            while ((x = pop()) != '(')
                postfix[j++] = x;
        }

        else {
            while (priority(stack[top]) >= priority(infix[i]))
                postfix[j++] = pop();
            push(infix[i]);
        }

        i++;
    }

    postfix[j] = '\0';

    printf("Postfix Expression: %s", postfix);

    return 0;
}
