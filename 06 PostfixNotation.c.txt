#include <stdio.h>

#define MAX 100

int stack[MAX];
int top = -1;

void push(int value) {
    top++;
    stack[top] = value;
}

int pop() {
    int value = stack[top];
    top--;
    return value;
}

int main() {
    char postfix[100];
    int i = 0, a, b, result;

    printf("Enter postfix expression: ");
    scanf("%s", postfix);

    while (postfix[i] != '\0') {

        // check if character is digit (without ctype.h)
        if (postfix[i] >= '0' && postfix[i] <= '9') {
            push(postfix[i] - '0');   // convert char to int
        }
        else {
            b = pop();
            a = pop();

            switch (postfix[i]) {
                case '+': result = a + b; break;
                case '-': result = a - b; break;
                case '*': result = a * b; break;
                case '/': result = a / b; break;
            }

            push(result);
        }

        i++;
    }

    printf("Result = %d\n", pop());

    return 0;
}
